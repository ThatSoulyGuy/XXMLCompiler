link goodbye.obj build\lib\Release\XXMLLLVMRuntime.lib ^
     kernel32.lib user32.lib ^
     libcmt.lib libvcruntime.lib libucrt.lib ^
     /OUT:goodbye.exe /SUBSYSTEM:CONSOLE /MACHINE:X64
